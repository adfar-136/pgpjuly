//themechanger 1
import React, { useState } from 'react'

export default function App() {
  const [bgColor,setBgColor] = useState("white")
  const [color,setColor] = useState("black")
  const [text,setText] = useState("Dark")
  const handleTheme=()=>{
    if(bgColor==="white"){
      setBgColor("black")
      setColor("white")
      setText("Light")
    } else {
      setBgColor("white")
      setColor("black")
      setText("Dark")
    }
  }
  return (
    <div style={{backgroundColor:bgColor,color:color,height:"100vh"}}>
      <h1>ThemeChanger</h1>
      <button onClick={handleTheme}>{text} Mode</button>
    </div>
  )
}

import React, { useEffect, useState } from 'react'
import Bitcoin from './components/Bitcoin'
import "./app.css"
export default function App() {
  const [data,setData] = useState("")
  async function handleData(){
     let data =await fetch("https://api.coinlore.net/api/tickers/")
     let jsonData = await data.json()
     setData(jsonData.data)
  }
  useEffect(()=>{
    handleData()
  },[])
  return (
    <div>
      <div className='main'>
        {data && (
          <>
           {data.map((item,index)=>{
            return (
                 <Bitcoin item={item} key={item.volume24}/>
            )
           })}
          </>
        )}
      </div>
    </div>
  )
}

import React from 'react'
import { BrowserRouter, Route, Routes } from 'react-router-dom'
import Home from './routes/Home'
import About from './routes/About'
import Contact from './routes/Contact'
import Blog from './routes/Blog'
import Navbar from './routes/Navbar'
import User from './routes/User'
import UserDetails from './routes/UserDetails'
import Adhaar from './routes/Adhaar';

export default function App() {
  
  return (
    <div>
     
      <BrowserRouter>
        <Navbar/>
        <Routes>
          <Route path='/' element={<Home/>}/>
          <Route path='/about' element={<About/>}/>
          <Route path='/contact' element={<Contact/>}/>
          <Route path='/blog' element={<Blog/>}/>
          <Route path='/about/users' element={<User/>}/>
          <Route path='/about/users/:id'
           element={<UserDetails/>}/>
          <Route path='/contact/:id/name/:adhaar'
          element={<Adhaar/>}/>
         
        </Routes>
      </BrowserRouter>
    </div>
  )
}
